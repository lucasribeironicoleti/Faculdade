create table produto(
	idproduto serial not null,
	estoque varchar(40),
	unidade varchar(40),
	constraint pk_produto primary key(idproduto)
);

create table tipobebida(
	idtipobebida serial not null,
	nometipobebida varchar(30),
	constraint pk_tipobebida primary key(idtipobebida)
);

create table bebida(
	idbebida serial not null,
	valor integer,
	litros integer,
	idproduto integer,
	idtipobebida integer,
	constraint pk_bebida primary key(idbebida),
	constraint fk_bebida_tipobebida foreign key(idtipobebida) references tipobebida(idtipobebida),
	constraint fk_bebida_produto foreign key(idproduto) references produto(idproduto) ON DELETE CASCADE
);
